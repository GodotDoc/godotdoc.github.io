    GDScript 语法参考 #

    GDScript 的语法很简单，下面通过一些案例帮助读者快速的入门 GDScript。

    本文默认读者至少有少许的编程经验，知道变量、常量、对象和类等词是什么意思。

    没有特殊说明，下文的代码都是在 func _ready(): 方法中编写的，然后点击右上角可以运行看效果。

    变量 #

    变量是用于存储信息的"容器"，这个"容器"储存的数据可以在游戏运行时随时更改，使用 var 关键定义。

    # 定义一个叫 n 的变量
var n = 1
# 使用冒号和数据类型关键字手动声明变量类型
var n1 : int = 1
# 可以省略类型关键字，系统会自动推断
var n2 : = 1

# 修改变量值
n = 2
n1 = 2
n2 = 2

    是否声明变量类型根据自己喜好来。笔者觉得声明变量类型使用起来比较方便，不声明类型代码看起来简洁。

    常量 #

    常量是在游戏运行时无法更改的值，使用关键字 const 定义。

    # 常量就是定义了就不能修改的数据"容器"
const A = 1

    注释 #

    注释是用来给程序员看的"代码"，不会被系统编译，相当于代码的备忘录，使用关键字 #。

    # 这里就是一条注释，可以理解为代码的备忘录，程序运行的时候，会把这行删掉。
var n = 1
const N = 1

    从 Godot4.2 开始，可以使用双井号 ## 注释。

    双井号注释被称为"文档注释"，可以像官方注释一样生成代码文档，也可以在检查器(Inspector)中鼠标悬停显示注释内容。

    extends Node2D

## 这是个变量的文档注释
@export var a = 1

## 这是个函数上的文档注释
func test_function():
	pass

    数据类型 #

    Godot 内置一些数据类型，分为基本数据类型(int、float、String、bool等)和对象数据类型(Dictionary、Array等)。基本数据类型在当作参数传递的时候是"按值传递"。对象数据类型在当作参数传递的时候是"引用传递"。

    数字类型 #

    整数类型 # 

    var n = 1
var n1 : int = 2

    int 存储的是 64 位值，等效于 C++ 中的 int64_t

    浮点数类型 # 

    var f = 1.1
var f1 :float = 1.2

    float 存储的是 64 位值，等效于 C++ 中的 double。

    字符类型 # 

    var s = "Hello Godot!"
var s1 : String = "Hello Godot!"

    默认使用 Unicode 字符。

    布尔类型 # 

    var b = false
var b1 : bool = true

    字典类型 # 

    var dict  =  {"key": "value", 2: 3}
var dict1 : Dictionary = {key = "value", other_key = 2}

# 返回该字典中键为 key 的值
var v1 = dict.get("key")
print(v1) # 输出 value

# 如果字典中没有名为 key1 的键，返回 null
var v2 = dict.get("key1")
print(v2) # 输出 null

# 不存在的时候返回某个默认值
var v3 = dict.get("key1","default value")
print(v3) # 输出 default value

# 合并两个字典
dict.merge(dict1) # 将 dict1 中的值合并到 dict 中
var v4 = dict.get("other_key")
print(v4) # 输出 2

# 移除键值
dict.erase("other_key")
var v5 = dict.get("other_key")
print(v5) # 输出 null

# 添加键值
dict["other_key"] = "new_value"
var v6 = dict.get("other_key")
print(v6) # 输出 new_value

# 更新某个键的值
dict["other_key"] = "updated_value"
var v7 = dict.get("other_key")
print(v7) # 输出 updated_value

    字典的值类型可以是任何类型。

    数组 # 

    var arr = [1, 2, 3]
var arr1 : Array = ["string1","string2","string3"]

# 添加值
arr.append(4)
print(arr) # 输出：[1, 2, 3, 4]

# 更新
arr[0] = -1 # 将索引为 0 位置的值改成 -1
print(arr) # 输出：[-1, 2, 3, 4]

# 取某个位置的值
var v1 = arr[1] 
print(v1) # 输出：2

# 移除某个值
arr.remove_at(0) # 移除索引为 0 位置的值
print(arr) # 输出：[2, 3, 4]
	
# 合并数组
arr.append_array(arr1)
print(arr) # 输出：[2, 3, 4, "string1", "string2", "string3"]

    数组设置类型 #

    上面的例子可以看出默认的 Array 类型是可以储存不同的数据类型，如果想只存某种类型可以使用“类型化数组”。

    # 只能存 int 类型
var a: Array[int]

# 只能存 Node 类型
var b: Array[Node]

# 只能存某种枚举类型
var d: Array[MyEnum]

    Array 等价于 Array[Varaint]

    压缩数组 #

    如果读者需要使用数组存储较多的值（数万个），建议使用压缩数组。压缩数组可以避免内存碎片，从而降低内存的使用量。但是压缩数组比普通的数组性能要低一点，所以不建议小数据集的时候使用压缩数组。

    var arr : PackedInt64Array = [1,2,3]
print(arr) # 输出：[1, 2, 3]

var arr1 : PackedStringArray = ["string1","string2","string3"]
print(arr1) # 输出：["string1", "string2", "string3"]
    • PackedByteArray：字节（从 0 到 255 的整数）数组。
    • PackedInt32Array：32位整数数组。
    • PackedInt64Array：64位整数数组。
    • PackedFloat32Array：32位浮点数数组。
    • PackedFloat64Array：64位浮点数数组。
    • PackedStringArray：字符串数组。
    • PackedVector2Array：Vector2 类型的数组。
    • PackedVector3Array：Vector3 类型的数组。
    • PackedColorArray：Color 类型的数组。

    运算 #

    数值运算 # 

    var n1 = 1
var n2 = 2

# 加
var n3 = n1 + n2
print(n3) # 输出 3

# 减
var n4 = n1 - n2
print(n4) # 输出 -1

# 乘
var n5 = n1 * n2
print(n5) # 输出 2

# 除
var n6 = 1.0 / 2.0
print(n6) # 输出 0.5
	
# 取余
var n7 = 10 % 3
print(n7) # 输出 1

    枚举 # 

    extends Node2D

# 匿名枚举类型
enum {UNIT_NEUTRAL, UNIT_ENEMY, UNIT_ALLY}

# 一个名称为 Name 的枚举
enum Name {THING_1, THING_2, ANOTHER_THING = -1}

func _ready():
	
	print(UNIT_NEUTRAL) # 输出：0
	
	print(Name.THING_1) # 输出：0
	
	print(Name.ANOTHER_THING) # 输出：-1

    枚举类型实际上存的的是就是数字类型，一些地方使用枚举类型可以提高性能和可维护性。

    循环语句 #

    程序中经常会用到迭代一个范围或者一个数组、字典类型的时候。

    # for 语句迭代数组
for i in [1,2,3,4,5]:
	print(i)
	# 输出 ： 
	# 1 
	# 2 
	# 3 
	# 4 
	# 5

# for 语句迭代数组
var arr = ['h','e','l','l','o']
for i in arr.size():
	print(arr[i])
	# 输出：
	# h
	# e
	# l
	# l
	# o

# for 语句迭代字典
var dic = {"a": "0", "b": "1", "c": "2"}
for i in dic:
	print(i + "=" + dic[i])
	# 输出
	# a=0
	# b=1
	# c=2

# for 语句迭代一定次数
for i in range(3):
	print(i)
	# 输出 ： 
	# 0 
	# 1 
	# 2 
# 其实 range(3) 返回的就是一个有三个元素的（[0,1,2]）的数组
# range 方法可以有很多玩法，读者感兴趣可以取看下 range 的文档。

# for 语句迭代字符
for c in "hello":
	print(c)
	# 输出：
	# h
	# e
	# l
	# l
	# o

    条件语句 #

    根据不同的条件执行不同的代码。

    If 语句 # 

    # if...else...
var x = 1
if x == 1:
	print("x = 1 时会执行这里")
else:
	print("x 不等于 1 时会执行这里")

# if...elif...
var x = 2
if x == 1:
	print("x = 1 时会执行这里")
elif x == 2:
	print("x 等于 2 时会执行这里")
else:
	print("x 不等于 1 且 x 不等于 2 时会执行这里")

# 类三运算符（这个还是挺常用的）
var x = 2
var y = 1111 if x == 1 else 1111
print(y)
# 输出：1111

    Match 语句 #

    默认情况下 Match 语句按照代码顺序，只要匹配到一个分支，下面分支就不会执行了。

    # 值匹配
var x = 1
match x:
	1:
		print("x = 1 时会执行这里")
	2:
		print("x = 2 时会执行这里")
	"test":
		print("x = test 时会执行这里")
	_:
		# 这里的 _ 相当于 default(默认值)，只有上面的都没匹配到才会走这里，可以省略。
		print("上面都没匹配会执行到这里")


# 多重匹配
var x = 1
match x:
	1,2,3,4:
		print("x 的值是 1或2或3或4")
	_:
		print("x 不等于 1，2，3，4 中的任何一个")


# 类型匹配
var x :int= 1
match typeof(x):
	TYPE_INT:
		print("x 是 int 类型的时候执行这里")
	TYPE_STRING:
		print("x 是 string 类型的时候执行这里")
	TYPE_ARRAY:
		print("x 是数组的时候执行这里")
	_:
		print("上面都没匹配会执行到这里")

# 绑定模式
var x = 3
match x:
	1:
		print("x = 1 时会执行这里")
	2:
		print("x = 2 时会执行这里")
	var new_x:
		# 只有上面都没匹配到才会走到这里，并将 x 的值赋给 new_x
		print(" new_x 的值为 ",new_x)
# 输出：new_x 的值为 3

    Match 数组

    var arr = [1,2,3] 
match arr:
	[]:
		# 当 arr 是空的时候执行这里
		print("arr 是空")
	[1,null]:
		# 当 arr 全等于 [1,null] 的时候执行这里
		print("arr = [1,null]")
	[var start, _, "test"]:
		# 当 数组长为 3 ，第二个值为人一直，第三个值为"test"的时候执行这里
		# 并且 start 会被复制数组的第一个值
		print(start)
	[1 , ..]:
		# 当数组的第一个值是 1 的时候执行这里
		print(" arr 的第一个值是 1")

    Match 字典

    var dic = {"name":"zhangsan","age":11}
match dic:
	{}:
		print("dic 是空的")
	{"name":"zhangsan"}:
		# dic 中只有一个 name 主键，并且对应的值是 zhangsan
		# 满足上面条件才会执行这里的代码
		print("dic 中 name 的值是 zhangsan")
	{"name": "zhangsan", "age": var age}:
		# dic 中只有 name 主键和 age 主键，且 name 对应的值为 zhangsan
		# 满足上面的条件才会执行这里的代码，且 age 的值为 dic 中 age 对应的值
		print("dic 中 age 的值是 ",age)
	{"name", "age"}:
		# dic 中只有 name 主键和 age 主键才会执行这里
		print("dic 中只含有 name 和 age 两个键值")
	{"name": "lisi", ..}:
		# dic 中只要含有 name 且对应的值为 lisi，才会走这里
		print("dic 中 name 的值为 lisi")

    类型转换 #

    如果需要将值强制转换为特定类型，特别是对于对象类型而言需要进行转型，则可以使用强制转型运算符 as。

    # 创建一个 sprite2D 类型
var sprite : Sprite2D = Sprite2D.new()
# 将上面的 sprite 转换成 Node2D 类型
var node2d : Node2D = sprite as Node2D

    类 #

    默认情况下，所有脚本文件都是未命名的类，这时只能使用文件的路径来引用这些无名类。

    MyClass.gd 文件

    extends Node

    Node2D.gd 文件

    extends Node2D

func _ready():
	var myclass = load("res://MyClass.gd")
	var myclassObj = myclass.new()
	print("success")

    可以通过 class_name 给自定义类起个类名，就可以通过类名创建对象了。

    MyClass.gd 文件

    class_name MyClass extends Node

    Node2D.gd 文件

    extends Node2D

func _ready():
	var myclassObj = MyClass.new()
	print("success")

    函数/方法 #

    函数始终属于类。查找变量时，函数作用域的查找顺序是：局部 → 类成员 → 全局。

    func _ready():
	var res = my_func(1,2)
	print(" x + y = ",res)
	# 输出： x + y = 3

# 自定义函数，函数名为 my_func 。
# 这个函数用于计算 x + y 
func my_func(x : int , y : int) -> int:
	return x + y

    上面的 my_func 是有返回值的，有时候需要不需要返回的的函数。

    func _ready():
	my_func(1,2)
	# 输出： x + y = 3

# 自定义函数，函数名为 my_func 。
# 这个函数用于计算 x + y 
func my_func(x : int , y : int) :
	var res = x + 1
	print(" x + y = ",res)

    参数 x 和 y 可以设置默认值。

    func _ready():
	# 此处只传第一个参数， x = 2 ，y 则取的默认值
	my_func(2)
	# 输出： x + y = 4

# 自定义函数，函数名为 my_func 。
# 这个函数用于计算 x + y 
func my_func(x : int = 1, y : int = 2) :
	var res = x + 1
	print(" x + y = ",res)

    引用函数 #

    引用函数就是将一个函数当作参数传给另一个函数，使用关键字 Callable。

    func _ready():
	var res = my_func(1,2,add_func)
	print("x + y = ",res)
	# 输出： x + y = 3

# 自定义函数，函数名为 my_func 。
func my_func(x : int , y : int , other_func : Callable) :
	# 直接调用 Callable 的 call() 方法，而不能直接用 ()
	return other_func.call(x,y)
	
# 将这个函数当作参数传给上面那个函数
func add_func(x : int , y : int) -> int:
	return x + y

    Lambda 函数 #

    lambda 函数可以用来创建不属于类的函数。比如创建一些 Callable 对象并当作参数传给其他函数的时候很有用。

    func _ready():
	# 定义lambda 函数
	var lambda_func = func (x : int , y : int):
		var res = x + y
		print("x + y = ",res)
	
	# 调用 lambda 函数
	lambda_func.call(1,2)
	# 输出：x + y = 3

    也可以个 lambda 函数设置个函数名。

    
func _ready():
	# 创建 lambda 函数，并给其设置个名字my_lambda_func
	var lambda_func = func my_lambda_func (x : int , y : int):
		var res = x + y
		print("x + y = ",res)
	
	# 调用 lambda 函数
	lambda_func.call(1,2)
	# 输出：x + y = 3

    lambda 函数不能是静态函数。

    静态函数 #

    函数可以声明为静态函数。静态函数不能访问实例成员变量，也不能使用 self，适用于创建辅助函数工具库。

    Utils.gd 文件

    class_name Utils extends Node

# 静态方法
static func my_sum_func(x : int , y :int) -> int:
	return x + y

    Node2D.gd 文件

    extends Node2D

func _ready():
	# 调用上面 Utils.gd 中定义的静态方法
	var res = Utils.my_sum_func(1,2)
	print("x + y = ",res)
	# 输出：x + y = 3
